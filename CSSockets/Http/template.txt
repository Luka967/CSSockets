- abstract public class HttpParser : BaseWritable
	-> public class HttpRequestParser : HttpParser
	-> public class HttpResponseParser : HttpParser
- public class HttpBodyParser : BaseDuplex

- abstract public class HttpBuilder : BaseReadable
	-> public class HttpRequestBuilder : HttpBuilder
	-> public class HttpResponseBuilder : HttpBuilder
- public class HttpBodyBuilder : BaseDuplex

- abstract public class HttpConnection
	- encapsulated public TcpSocket
	- encapsulated public HttpBodyParser
	- encapsulated public HttpBodyBuilder
	- piping: HttpBodyBuilder => TcpSocket => HttpBodyParser
	-> public class HttpClientConnection : HttpConnection
		- encapsulated private HttpResponseParser
		- encapsulated private HttpRequestBuilder
		- piping: HttpRequestBuilder => HttpClientConnection => HttpResponseParser
	-> public class HttpServerConnection : HttpConnection
		- encapsulated private HttpRequestParser
		- encapsulated private HttpResponseBuilder
		- piping: HttpResponseBuilder => HttpServerConnection => HttpRequestParser

- public class HttpListener
	- encapsulated public TcpListener
